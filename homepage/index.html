<script src="https://cdn.moduengine.com/modu.min.js"></script>
<script>
const canvas = document.getElementById('game');
const game = createGame();
game.addPlugin(Simple2DRenderer, canvas);

game.defineEntity('player')
    .with(Transform2D)
    .with(Sprite, { shape: SHAPE_CIRCLE, radius: 20 })
    .with(Player)
    .register();

const input = game.addPlugin(InputPlugin, canvas);
input.action('move', { type: 'vector', bindings: ['wasd'] });

game.addSystem(() => {
    for (const p of game.query('player')) {
        const dir = game.world.getInput(p.get(Player).clientId)?.move;
        if (dir) p.setVelocity(dir.x * 5, dir.y * 5);
    }
});

game.connect('my-room', {
    onConnect(id) {
        game.spawn('player', {
            x: dRandom() * 800,
            y: dRandom() * 600,
            clientId: id
        });
    }
});
</script>

Why Modu?

Instant Play

Your game runs locally immediately. No "connecting..." screens. When you go online, you're already playing.

One Codebase

No separate single-player or multiplayer modes. The same code works for 1 player or 100.

Deterministic Sync

All clients run the same simulation. Same inputs = same state. No rubberbanding, no desync.

No Server Logic

Server just relays inputs. No game code on the server. Deploy anywhere, scale infinitely.

ECS Architecture

Entity-Component-System design. Compose behaviors, query efficiently, think in data.

Built-in Debug Tools

Determinism guard warns about unsafe code. Debug UI shows sync status. Find bugs before players do.

How It Works

1

Player presses a key

Input captured locally

→
2

Input sent to server

Server assigns sequence number

→
3

Broadcast to all clients

Everyone gets same inputs

→
4

Same simulation runs

Identical state everywhere

The server never runs game logic. It's just a message broker.
Your game code runs on clients. Determinism ensures they agree.

Simple API

Define Entities

game.defineEntity('bullet')
    .with(Transform2D)
    .with(Body2D, {
        shapeType: SHAPE_CIRCLE,
        radius: 5
    })
    .with(Sprite, {
        shape: SHAPE_CIRCLE,
        color: '#ff0'
    })
    .register();

Write Systems

game.addSystem(() => {
    for (const b of game.query('bullet')) {
        // Destroy bullets that go off screen
        if (b.get(Transform2D).y > 1000) {
            b.destroy();
        }
    }
});

Handle Collisions

physics.onCollision('bullet', 'enemy',
    (bullet, enemy) => {
        enemy.destroy();
        bullet.destroy();
        game.spawn('explosion', {
            x: enemy.get(Transform2D).x,
            y: enemy.get(Transform2D).y
        });
    }
);
